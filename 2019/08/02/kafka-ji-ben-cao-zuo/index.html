Kafka 基本操作

创建topic

./kafka-topics.sh --create --zookeeper host0:2181,host1:2181,host2:2181,host3:2181,host4:2181 --replication-factor 1 --partitions 1 --topic test

#-replication-factor 1 数据副本数为1
#-partitions 1 分区数量
#--topic test topic name

查看topic

./kafka-topics.sh --list  --zookeeper host0:2181,host1:2181,host2:2181,host3:2181,host4:2181

生产消息

./kafka-console-producer.sh --broker-list host1:6667,host4:6667,host3:6667,host2:6667 --topic test
#--broker-list 为kafka broker所有节点的列表（集群版本）
#host1:6667 为kafka配置中listeners的配置端口号
#--topic test    消息发送的topic name

消费消息

./kafka-console-consumer.sh --zookeeper host0:2181,host1:2181,host2:2181,host3:2181,host4:2181 --topic test --from-beginning
